CREATE TABLE ADMINBOARD 
(
  B_ID NUMBER NOT NULL 
, TITLE VARCHAR2(300 BYTE) NOT NULL 
, CONTENT VARCHAR2(4000 BYTE) NOT NULL 
, LIKE_CNT NUMBER DEFAULT 0 NOT NULL 
, B_DATE DATE DEFAULT sysdate NOT NULL 
, U_DATE DATE DEFAULT sysdate NOT NULL 
, CONSTRAINT ADMINBOARD_PK PRIMARY KEY (
    B_ID 
  )
);


CREATE TABLE AREA 
(
  A_ID VARCHAR2(10 BYTE) NOT NULL 
, AREA VARCHAR2(30 BYTE) NOT NULL 
, CONSTRAINT AREA_PK PRIMARY KEY 
  (
    A_ID 
  )
);


CREATE TABLE CATEGORY 
(
  CATE_ID VARCHAR2(10 BYTE) NOT NULL 
, CATEGORY VARCHAR2(30 BYTE) NOT NULL 
, CONSTRAINT CATEGORY_PK PRIMARY KEY 
  (
    CATE_ID 
  )
);


CREATE TABLE FAVORITE 
(
  B_ID NUMBER 
, ID VARCHAR2(20 BYTE) 
);


CREATE TABLE MEMBER 
(
  ID VARCHAR2(20 BYTE) NOT NULL 
, PW VARCHAR2(30 BYTE) NOT NULL 
, EMAIL VARCHAR2(50 BYTE) NOT NULL 
, PHONE VARCHAR2(15 BYTE) NOT NULL 
, CONSTRAINT SYS_C007899 PRIMARY KEY 
  (
    ID 
  )
);

CREATE TABLE NATION 
(
  N_ID VARCHAR2(10 BYTE) NOT NULL 
, NATION VARCHAR2(30 BYTE) NOT NULL 
, A_ID VARCHAR2(10 BYTE) NOT NULL 
, LAT NUMBER 
, LNG NUMBER 
, CONSTRAINT NATION_PK PRIMARY KEY 
  (
    N_ID 
  )
);


CREATE TABLE REPLY 
(
  R_ID NUMBER NOT NULL 
, B_ID NUMBER NOT NULL 
, ID VARCHAR2(20 BYTE) NOT NULL 
, CONTENT VARCHAR2(500 BYTE) NOT NULL 
, R_DATE DATE DEFAULT sysdate NOT NULL 
, U_DATE DATE DEFAULT sysdate NOT NULL 
, CONSTRAINT REPLY_PK PRIMARY KEY 
  (
    R_ID 
  )
);


CREATE TABLE USERBOARD 
(
  B_ID NUMBER NOT NULL 
, ID VARCHAR2(20 BYTE) NOT NULL 
, TITLE VARCHAR2(300 BYTE) NOT NULL 
, CONTENT VARCHAR2(4000 BYTE) 
, R_CNT NUMBER DEFAULT 0 NOT NULL 
, LIKE_CNT NUMBER DEFAULT 0 NOT NULL 
, B_DATE DATE DEFAULT SYSDATE NOT NULL 
, U_DATE DATE DEFAULT SYSDATE 
, B_TYPE VARCHAR2(10 BYTE) NOT NULL 
, A_ID VARCHAR2(10 BYTE) 
, N_ID VARCHAR2(10 BYTE) 
, CATE_ID VARCHAR2(10 BYTE) 
, CONSTRAINT USERBOARD_PK PRIMARY KEY 
  (
    B_ID 
  )
);


CREATE SEQUENCE REPLY_SEQ START WITH 1 INCREMENT BY 1 MAXVALUE 10000 CYCLE NOCACHE;


CREATE SEQUENCE USERBOARD_SEQ START WITH 1000 INCREMENT BY 1 MAXVALUE 10000 CYCLE NOCACHE;
